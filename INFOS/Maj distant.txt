git rev-parse --is-inside-work-tree > /dev/null

echo "➡️ Passage sur la branche main..."
git checkout main

echo "➡️ Ajout de tous les fichiers..."
git add -A

echo "➡️ Commit forcé (si nécessaire)..."
git commit -m "Force update main from local workspace" || echo "ℹ️ Aucun changement à committer"

echo "⚠️ PUSH FORCE vers origin/main..."
git push origin main --force

echo "✅ Mise à jour distante TERMINÉE."

🟢 Script STANDARD (recommandé)

➡️ Commit + push sans forcer (sécurisé)

📄 push_main.sh
#!/bin/bash

# Arrêt immédiat en cas d’erreur
set -e

echo "=== Déploiement vers GitHub /main ==="

# 1. Se placer dans le dépôt
cd /c/dev/elections-municipales || {
  echo "❌ Répertoire introuvable"
  exit 1
}

# 2. Vérifier la branche
CURRENT_BRANCH=$(git branch --show-current)
if [ "$CURRENT_BRANCH" != "main" ]; then
  echo "❌ Tu n'es pas sur la branche main (actuelle : $CURRENT_BRANCH)"
  exit 1
fi

# 3. Afficher l’état
echo "=== État du dépôt ==="
git status

# 4. Ajouter tous les fichiers
git add .

# 5. Commit (message horodaté)
COMMIT_MSG="Update $(date '+%Y-%m-%d %H:%M')"
git commit -m "$COMMIT_MSG" || echo "ℹ️ Rien à committer"

# 6. Push
git push origin main

echo "✅ Push terminé avec succès"

🔴 Script FORCÉ (à utiliser en connaissance de cause)

➡️ Écrase origin/main avec ton local

📄 push_main_force.sh
#!/bin/bash

set -e

echo "⚠️ PUSH FORCÉ VERS origin/main ⚠️"

cd /c/dev/elections-municipales || exit 1

CURRENT_BRANCH=$(git branch --show-current)
if [ "$CURRENT_BRANCH" != "main" ]; then
  echo "❌ Tu n'es pas sur main"
  exit 1
fi

git status
read -p "❗ Confirmer le push forcé ? (oui/NON) : " CONFIRM
if [ "$CONFIRM" != "oui" ]; then
  echo "❌ Annulé"
  exit 0
fi

git add .
git commit -m "Force update $(date '+%Y-%m-%d %H:%M')" || true
git push origin main --force

echo "✅ Push forcé terminé"
